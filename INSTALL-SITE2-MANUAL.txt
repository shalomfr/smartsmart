התקנה ידנית של site2
======================

אם הסקריפטים האוטומטיים לא עובדים, הנה הוראות להתקנה ידנית:

1. התחבר לשרת:
   ssh root@31.97.129.5

2. הרץ את הפקודות האלה אחת אחת:

# יצירת תיקייה
mkdir -p /home/emailapp/site2
cd /home/emailapp/site2

# העתקת קוד
cp -r /home/emailapp/email-app/* .

# התקנת Frontend
npm install
npm run build

# התקנת Backend
cd backend
echo "PORT=4000" > .env
npm install

# הפעלת שירותים
pm2 delete site2-backend site2-frontend
pm2 start server.js --name site2-backend
cd ..
pm2 serve dist 9000 --name site2-frontend --spa
pm2 save

# הגדרת Nginx - צור קובץ חדש
nano /etc/nginx/sites-available/site2

# הדבק את התוכן הבא:
server {
    listen 8081;
    server_name _;
    
    location / {
        proxy_pass http://localhost:9000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
    
    location /api {
        proxy_pass http://localhost:4000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

# שמור עם Ctrl+X, Y, Enter

# הפעל את האתר
ln -sf /etc/nginx/sites-available/site2 /etc/nginx/sites-enabled/
nginx -t && systemctl reload nginx

# פתח פורטים
ufw allow 4000/tcp
ufw allow 9000/tcp
ufw allow 8081/tcp
ufw reload

# בדוק שהכל עובד
pm2 list
netstat -tlnp | grep -E "4000|9000|8081"

3. האתר אמור להיות זמין ב:
   http://31.97.129.5:8081
